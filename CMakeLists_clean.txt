cmake_minimum_required(VERSION 3.16)
project(CHTLCompiler VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-unused-parameter -Wno-unused-variable")

include_directories(src)

# 🎯 按照正确的架构组织：4个编译器 + Scanner + Dispatcher + Common

# 统一扫描器
add_library(chtl_scanner STATIC
    src/Scanner/CHTLUnifiedScanner.cpp
)

# 编译器调度器
add_library(chtl_dispatcher STATIC
    src/Dispatcher/CompilerDispatcher.cpp
)
target_link_libraries(chtl_dispatcher chtl_scanner)

# 共同组件
add_library(chtl_common STATIC
    src/Common/Core/NamespaceAlgorithms.cpp
    src/Common/Core/ErrorHandler.cpp
    src/Common/Utils/FileUtils.cpp
    src/Common/ModulePathManager.cpp
)

# CHTL编译器（完整体系）
add_library(chtl_compiler STATIC
    src/CHTLCompiler/Import/ImportManager.cpp
    # 其他CHTL编译器文件根据需要添加
)
target_link_libraries(chtl_compiler chtl_common)

# CHTL JS编译器（完整体系）
add_library(chtljs_compiler STATIC
    src/CHTLJSCompiler/Animation/AnimateSystemProcessor.cpp
    src/CHTLJSCompiler/CHTLJSCompiler.cpp
    # 其他CHTL JS编译器文件根据需要添加
)
target_link_libraries(chtljs_compiler chtl_common)

# CSS编译器
add_library(css_compiler STATIC
    src/CSSCompiler/Core/CSSCompilerCore.cpp
)
target_link_libraries(css_compiler chtl_common)

# JavaScript编译器
add_library(js_compiler STATIC
    src/JavaScriptCompiler/Core/JavaScriptCompilerCore.cpp
)
target_link_libraries(js_compiler chtl_common)

# 主程序测试
add_executable(chtl_main src/main.cpp)
target_link_libraries(chtl_main 
    chtl_scanner 
    chtl_dispatcher 
    chtl_compiler 
    chtljs_compiler 
    css_compiler 
    js_compiler 
    chtl_common
)

# 简单的Scanner测试
add_executable(test_scanner_clean test/test_scanner_clean.cpp)
target_link_libraries(test_scanner_clean chtl_scanner)

# 设置输出目录
set_target_properties(
    chtl_scanner chtl_dispatcher chtl_common chtl_compiler chtljs_compiler css_compiler js_compiler
    PROPERTIES ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib
)

set_target_properties(
    chtl_main test_scanner_clean
    PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
