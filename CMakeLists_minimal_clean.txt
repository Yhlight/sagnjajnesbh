cmake_minimum_required(VERSION 3.16)
project(CHTLCompilerClean VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-unused-parameter -Wno-unused-variable")

include_directories(src)

# 🎯 清理后的简单架构测试：仅Scanner
add_library(chtl_scanner_clean STATIC
    src/Scanner/CHTLUnifiedScanner.cpp
)

# Scanner测试
add_executable(test_clean_architecture test/test_scanner_clean.cpp)
target_link_libraries(test_clean_architecture chtl_scanner_clean)

# 设置输出目录
set_target_properties(chtl_scanner_clean PROPERTIES ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set_target_properties(test_clean_architecture PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
