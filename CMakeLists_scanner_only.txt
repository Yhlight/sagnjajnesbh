cmake_minimum_required(VERSION 3.16)
project(CHTLScanner VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-unused-parameter -Wno-unused-variable")

include_directories(src)

# 仅测试统一扫描器
add_library(chtl_scanner_only STATIC
    src/UnifiedCore/Scanner/CHTLUnifiedScanner.cpp
)

# 测试统一扫描器
add_executable(test_scanner_only test/test_scanner_only.cpp)
target_link_libraries(test_scanner_only chtl_scanner_only)

# 设置输出目录
set_target_properties(chtl_scanner_only PROPERTIES ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set_target_properties(test_scanner_only PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
